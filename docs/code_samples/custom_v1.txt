import com.mindee.MindeeClient;
import com.mindee.input.LocalInputSource;
import com.mindee.parsing.common.Document;
import com.mindee.product.custom.CustomV1Inference;
import com.mindee.http.CustomEndpoint;
import java.io.File;
import java.io.IOException;

public class SimpleMindeeClient {

  public static void main(String[] args) throws IOException {
    String apiKey = "my-api-key";
    String filePath = "/path/to/the/file.ext";

    MindeeClient mindeeClient = new MindeeClient(apiKey);
    LocalInputSource inputSource = new LocalInputSource(new File(filePath));

    CustomEndpoint customEndpoint = new CustomEndpoint(
        "my-endpoint",
        "my-account"
    );
    Document<CustomV1Inference> documentParsed =  mindeeClient.parse(
        inputSource,
        customEndpoint
    );

    System.out.println(documentParsed.toString());
  }

}
